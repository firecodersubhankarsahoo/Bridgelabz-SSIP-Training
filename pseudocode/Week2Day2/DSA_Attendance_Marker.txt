Pseudo-code Problem: Attendance Marker
Concepts: 2D Arrays, Iteration, Search

matrix[30][30]  # 30 students x 30 days

method markAttendance(studentIndex, dayIndex, value)
    matrix[studentIndex][dayIndex] = value

method dailySummary(dayIndex)
    count = 0
    for i=0 to 29
        if matrix[i][dayIndex] == 1
            count++
    return count

method monthlySummary(studentIndex)
    count = 0
    for j=0 to 29
        if matrix[studentIndex][j] == 1
            count++
    return count

method defaulterList(threshold=22)
    list = []
    for i=0 to 29
        total = monthlySummary(i)
        if total < threshold
            add i to list
    return list
