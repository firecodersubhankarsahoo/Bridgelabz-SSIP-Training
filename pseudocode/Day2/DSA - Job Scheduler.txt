Pseudo-code for DSA - Job Scheduler
--------------------------------------------------
CLASS Job
    PRIVATE id : STRING
    PRIVATE deadline : INTEGER
    PRIVATE processingTime : INTEGER
END CLASS

CLASS JobScheduler
    PRIVATE pq : PRIORITY_QUEUE by earliest deadline

    METHOD addJob(job : Job)
        INSERT job INTO pq
    END METHOD

    METHOD processJobs()
        WHILE pq NOT EMPTY
            job = REMOVE pq
            PRINT "Processing job " + job.id
        END WHILE
    END METHOD
END CLASS

MAIN
    scheduler = NEW JobScheduler()
    scheduler.addJob(Job("J1", 2, 5))
    scheduler.addJob(Job("J2", 1, 3))
    scheduler.processJobs()
END MAIN
