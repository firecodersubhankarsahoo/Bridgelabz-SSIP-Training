CLASS BusRoute
    ATTRIBUTE: linkedList stops
    METHOD addAfter(existingStop, newStop)
        FIND existingStop in linkedList
        INSERT newStop immediately after it
    METHOD remove(stop)
        DELETE stop if exists
    METHOD contains(stop)
        RETURN true if stop found
