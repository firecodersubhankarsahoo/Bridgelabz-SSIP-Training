CLASS TaskManager
    ATTRIBUTES:
        LinkedHashMap tasks
        nextId = 1
    METHOD addTask(desc)
        tasks[nextId] = desc
        nextId++
    METHOD removeTask(id)
        REMOVE entry with key=id
    METHOD viewTasks()
        PRINT all tasks in insertion order
